# 对链表进行插入排序





```js
function insertionSortList(head){
    var dummy = new ListNode(0);
    var pre = dummy;
		while(head) {
      var cur = head;
      head = head.next;
      if(cur.val < pre.val) {
        pre = dummy;
      }
      while(pre.next !== null && cur.val > pre.next.val){
        pre = pre.next;
      }
       cur.next = pre.next;
       pre.next = cur;
    }
    return dummy.next;
}
```

```js
public ListNode insertionSortList(ListNode head) {
     if (head == null) {
       return head;
     }
         
     const helper = new ListNode(0);
     const cur = head;
     const pre = helper;
     const next = null;
     while (cur != null) {
        next = cur.next;
        while (pre.next != null && pre.next.val < cur.val) {
            pre = pre.next;
        }
        cur.next = pre.next;
        pre.next = cur;
        pre = helper;
        cur = next;
    }
    return helper.next;
}
```
